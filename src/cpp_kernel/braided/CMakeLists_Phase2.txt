cmake_minimum_required(VERSION 3.14)
project(BraidedRSE_Phase2)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Include directories
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/..)

# Phase 2 Test Executable
add_executable(test_phase2
    ../demos/test_phase2.cpp
)

# Link libraries
target_link_libraries(test_phase2
    pthread
    atomic
)

# Compiler flags
target_compile_options(test_phase2 PRIVATE
    -Wall
    -Wextra
    -O2
)

message(STATUS "Configured Phase 2 test build")
